2.鼠标点击特效

代码：

$(function(){
    // 为网页添加一个点击特效：随机显示文字，随机显示颜色
    $("body").click(function(e) {

        // 创建两个数组
        var word= new Array("富强", "民主", "文明", "和谐", "自由", "平等", "公正" ,"法治", "爱国", "敬业", "诚信", "友善");
        var color = new Array("#ffcc00", "#b870d6", "#89ddf3");

        // 创建一个a标签,随机从word数组中选一个写入a中
        var $i = $("<a href='#' style='text-decoration:none'/>").text(word[Math.floor(Math.random()*word.length)]);

        // 获取鼠标点击的位置x，y
        var x = e.pageX,y = e.pageY;
        
        // 设置a标签样式
        // 注意：一定要设置定位
        $i.css({
            "z-index": 99999999,
            "top": y - 20,
            "left": x,
            "position": "absolute",
            "font-weight": "bold",
            "color": color[Math.floor(Math.random()*color.length)]
        });

        // 将a标签添加到body
        $("body").append($i);

        // 设置a标签1.5秒后消失
        $i.animate({
            "top": y - 180,
            "opacity": 0
        },
        1500,
        function() {
            $i.remove();
        });
    });
})
